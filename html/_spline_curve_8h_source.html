32 /***********************************************************************
33 The indexing scheme for control points and knots is the one used in
34 Gerald Farin's book "Curves and Surfaces for CAGD." That is, the domain
35 of a single B-spline basis function of degree n, N_i^n(u), is the half-
36 open interval dom(N_i^n(u)) = [u_{i-1}, u_{i+n}). Then a B-spline of
37 degree n with p control points has a control point vector C = (c_0, c_1,
38 ..., c_{p-1}) and a knot vector U = (u_{-1}, u_0, ..., u_{p-1+n}). Since
39 the first and last knots, u_{-1} and u_{p-1+n}, are never referenced in
40 deBoor's algorithm, they are not passed to the SplineCurve constructor
41 and not stored in the knot array. In other words, the knot array only
42 contains the p-1+n knots from u_0 to u_{p-2+n}.
43 Sorry about the confusion.
44 ***********************************************************************/

90  int findInterval(Scalar u) const // Localizes the knot interval containing u (knots[i]<=u<knots[i+1])
91  {
92  /* Perform binary search on the valid range of the knot vector: */
93  int l=degree-1;
94  int r=numPoints-1;
95  while(r-l>1)
96  {
97  int t=(l+r)>>1;
98  if(u<knots[t])
99  r=t;
100  else
101  l=t;
102  }
103 
104  return l;
105  }
106  void deBoorStage(Scalar u,EvaluationCache* cache,int iv,int k) const // Performs a single stage of deBoor's algorithm
107  {
108  int subDegree=degree-k;
109  const Scalar* knotBase=&knots[iv-subDegree+1];
110  for(int i=0;i<subDegree;++i,++knotBase)
111  {
112  Scalar alpha=(u-knotBase[0])/(knotBase[subDegree]-knotBase[0]);
113  cache->points[i]=affineCombination(cache->points[i],cache->points[i+1],alpha);
114  }
115  }
