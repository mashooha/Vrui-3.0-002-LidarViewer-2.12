73  return P(m(0,0)*p[0]+m(0,1)*p[1]+m(0,2),

75  }

77  {
78  return HV(m(0,0)*hv[0]+m(0,1)*hv[1]+m(0,2)*hv[2],
79  m(1,0)*hv[0]+m(1,1)*hv[1]+m(1,2)*hv[2],
80  hv[2]);
81  }
82  inline static V inverseTransform(const M& m,const V& v)
83  {
84  double det=double(m(0,0))*double(m(1,1))-double(m(1,0))*double(m(0,1));
85  return V(ScalarParam((double(m(1,1))*double(v[0])-double(m(0,1))*double(v[1]))/det),
86  ScalarParam((double(m(0,0))*double(v[1])-double(m(1,0))*double(v[0]))/det));
87  }
88  inline static P inverseTransform(const M& m,const P& p)
89  {
90  double det=double(m(0,0))*double(m(1,1))-double(m(1,0))*double(m(0,1));
91  return P(ScalarParam((double(m(1,1))*(double(p[0])-double(m(0,2)))-double(m(0,1))*(double(p[1])-double(m(1,2))))/det),
92  ScalarParam((double(m(0,0))*(double(p[1])-double(m(1,2)))-double(m(1,0))*(double(p[0])-double(m(0,2))))/det));
93  }
94  inline static HV inverseTransform(const M& m,const HV& hv)
95  {
96  double det=double(m(0,0))*double(m(1,1))-double(m(1,0))*double(m(0,1));
97  return HV(ScalarParam((double(m(1,1))*(double(hv[0])-double(m(0,2))*double(hv[2]))-double(m(0,1))*(double(hv[1])-double(m(1,2))*double(hv[2])))/det),
98  ScalarParam((double(m(0,0))*(double(hv[1])-double(m(1,2))*double(hv[2]))-double(m(1,0))*(double(hv[0])-double(m(0,2))*double(hv[2])))/det),
99  hv[2]);
