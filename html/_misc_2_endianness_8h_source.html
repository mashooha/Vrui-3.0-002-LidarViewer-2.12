47 /****************************************************************
48 Helper class to allow partial specialization of endianness
49 swapper:
50 ****************************************************************/
51 
52 template <class ValueParam>
53 class EndiannessSwapper
54  {
55  /* Methods: */
56  public:
57  static void swap(ValueParam& value)
58  {
59  /* Treat value as array of bytes: */
60  unsigned char* bytes=reinterpret_cast<unsigned char*>(&value);
61 
62  /* Swap value byte by byte: */
63  size_t i1,i2;
64  for(i1=0,i2=sizeof(ValueParam)-1;i1<i2;++i1,--i2)
65  {
66  unsigned char temp=bytes[i1];
67  bytes[i1]=bytes[i2];
68  bytes[i2]=temp;
69  }
70  }
71  static void swap(ValueParam* values,size_t numValues)
72  {
73  for(size_t i=0;i<numValues;++i)
74  {
75  /* Treat value as array of bytes: */
76  unsigned char* bytes=reinterpret_cast<unsigned char*>(values+i);
77 
78  /* Swap value byte by byte: */
79  size_t i1,i2;
80  for(i1=0,i2=sizeof(ValueParam)-1;i1<i2;++i1,--i2)
81  {
82  unsigned char temp=bytes[i1];
83  bytes[i1]=bytes[i2];
84  bytes[i2]=temp;
85  }
86  }
87  }
88  };
