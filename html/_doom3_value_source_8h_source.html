62  void skipComments(void) // Skip newlines and line or block comments in the character source
63  {
64  while(Base::peekc()=='\n'||Base::peekc()=='/')
65  {
66  if(Base::peekc()=='\n')
67  {
68  /* Skip the newline: */
69  Base::getChar();
70  ++lineNumber;
71  Base::skipWs();
72  }
73  else
74  {
75  /* Skip the slash: */
76  Base::getChar();
77 
78  /* Check the next character: */
79  if(Base::peekc()=='/') // Line comment
80  {
81  /* Skip to the end of the line: */
82  Base::skipLine();
83  ++lineNumber;
84  Base::skipWs();
85  }
86  else if(Base::peekc()=='*') // Block comment
87  {
88  Base::getChar();
89  skipBlockComment();
90  Base::skipWs();
91  }
92  else // Not a comment after all
93  {
94  /* Return the slash to the character source: */
95  Base::ungetChar('/');
96  break;
97  }
98  }
99  }
