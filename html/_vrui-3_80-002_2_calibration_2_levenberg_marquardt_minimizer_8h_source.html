58  /* Initialize the optimizer: */
59  Scalar tau=Scalar(1.0e-3);
60  Scalar epsilon1=Scalar(1.0e-20);
61  Scalar epsilon2=Scalar(1.0e-20);
62  int maxNumIterations=1000;
63 
64  /* Compute the Jacobian matrix, the error vector, and the initial target function value: */
65  Matrix A;
66  Vector g;
67  for(int i=0;i<dimension;++i)
68  {
69  for(int j=0;j<dimension;++j)
70  A(i,j)=Scalar(0);
71  g[i]=Scalar(0);
72  }
73  Scalar F(0);
74  for(size_t index=0;index<fitter.getNumPoints();++index)
75  {
76  Derivative dp=fitter.calcDistanceDerivative(index);
77  Scalar d=fitter.calcDistance(index);
78  for(int i=0;i<dimension;++i)
79  {
80  for(int j=0;j<dimension;++j)
81  A(i,j)+=dp[i]*dp[j];
82  g[i]+=dp[i]*d;
83  }
84  F+=Math::sqr(d);
85  }
86  F*=Scalar(0.5);
87 
88  /* Compute the initial damping factor: */
89  Scalar maxA=A(0,0);
90  for(int i=1;i<dimension;++i)
91  if(maxA<A(i,i))
92  maxA=A(i,i);
93  Scalar mu=tau*maxA;
94  Scalar nu=Scalar(2);
95 
96  /* Check for convergence: */
97  bool found=true;
98  for(int i=0;i<dimension;++i)
99  if(Math::abs(g[i])>epsilon1)
100  found=false;
101  int iteration;
102  for(iteration=0;!found&&iteration<maxNumIterations;++iteration)
103  {
104  /* Calculate step direction: */
105  Matrix H=A;
106  for(int i=0;i<dimension;++i)
107  H(i,i)+=mu;
108  Vector h=g/H; // h is actually the negative of hlm in the pseudo-code
109 
110  /* Check for convergence: */
111  if(Geometry::mag(h)<=epsilon2*(fitter.calcMag()+epsilon2))
112  break;
113 
114  /* Try updating the current state: */
115  fitter.save();
116  fitter.increment(h); // Subtracts h instead of adding (h is negative, see above)
117  fitter.normalize();
118 
119  /* Calculate the new target function value: */
120  Scalar newF(0);
121  for(size_t index=0;index<fitter.getNumPoints();++index)
122  newF+=Math::sqr(fitter.calcDistance(index));
123  newF*=Scalar(0.5);
124

127  for(int i=0;i<dimension;++i)
128  denom+=h[i]*(mu*h[i]+g[i]); // Adds g instead of subtracting (h is negative, see above)
129  denom*=Scalar(0.5);
130  Scalar rho=(F-newF)/denom;
131 
132  /* Accept or deny the step: */
133  if(rho>Scalar(0))
134  {
135  /* Compute the new Jacobian matrix and the new error vector: */
136  for(int i=0;i<dimension;++i)
137  {
138  for(int j=0;j<dimension;++j)
139  A(i,j)=Scalar(0);
140  g[i]=Scalar(0);
141  }

143  {

145  Scalar d=fitter.calcDistance(index);
146  for(int i=0;i<dimension;++i)
147  {
148  for(int j=0;j<dimension;++j)
149  A(i,j)+=dp[i]*dp[j];
150  g[i]+=dp[i]*d;
151  }
152  }
153 
154  /* Update the target function value: */
155  F=newF;
156 
157  /* Check for convergence: */
158  found=true;
159  for(int i=0;i<dimension;++i)
160  if(Math::abs(g[i])>epsilon1)
161  found=false;
162

164  Scalar rhof=Scalar(2)*rho-Scalar(1);
165  Scalar factor=Scalar(1)-rhof*rhof*rhof;
166  if(factor<Scalar(1)/Scalar(3))
167  factor=Scalar(1)/Scalar(3);
168  mu*=factor;
169  nu=Scalar(2);
170  }
171  else

179  }
