74  /*********************************************************************
75  Conversions between Geoid-centered Geoid-fixed Cartesian and geodetic
76  (longitude/latitude/elevation) coordinates:
77  Geodetic points are in (longitude, latitude, elevation) in radians and
78  whatever unit was used for the Geoid's radius.
79  Cartesian points are (x, y, z) where (0, 0, 0) is the Geoid's
80  centroid, the north pole is at (0, 0, +b), and the 0 meridian is in
81  the y=0 plane.
82  Cartesian frames have their z axes normal to the ellipsoid pointing to
83  the outside (away from the centroid), and their y axes pointing north
84  along a meridian.
85  *********************************************************************/

87  Point geodeticToCartesian(const Point& geodetic) const // Transforms a point
88  {
89  double sLon=Math::sin(double(geodetic[0]));
90  double cLon=Math::cos(double(geodetic[0]));
91  double sLat=Math::sin(double(geodetic[1]));
92  double cLat=Math::cos(double(geodetic[1]));
93  double elev=double(geodetic[2]);
94  double chi=Math::sqrt(1.0-e2*sLat*sLat);
95  return Point(Scalar((radius/chi+elev)*cLat*cLon),Scalar((radius/chi+elev)*cLat*sLon),Scalar((radius*(1.0-e2)/chi+elev)*sLat));
96  }
