88  static ControlPoint affineCombination(const ControlPoint& cp1,const ControlPoint& cp2,Scalar w2) // Returns affine combination of two control points
89  {
90  ControlPoint result;
91  result.center=Geometry::affineCombination(cp1.center,cp2.center,w2);
92  Scalar w1=Scalar(1)-w2;
93  result.size=cp1.size*w1+cp2.size*w2;
94  result.forward=cp1.forward*w1+cp2.forward*w2;
95  result.forward.normalize();
96  result.up=cp1.up*w1+cp2.up*w2;
97  result.up.normalize();
98  return result;
99  }
