148  static void interpolate(const Interpolant& i0,const Interpolant& i1,Scalar w1,Interpolant& result)
149  {
150  Interpolant delta=invert(i0);
151  delta.leftMultiply(i1);
152  const Scalar* q=delta.getQuaternion();
153  result=i0;
154  if(Math::abs(q[3])<Scalar(1))
155  {
156  Scalar alpha=Math::acos(q[3]);
157  Scalar f=Math::sin(w1*alpha)/Math::sin(alpha);
158  double qn[4];
159  for(int i=0;i<3;++i)
160  qn[i]=q[i]*f;
161  qn[3]=Math::cos(w1*alpha);
162  result*=Interpolant::fromQuaternion(qn);
163  }
164  }
