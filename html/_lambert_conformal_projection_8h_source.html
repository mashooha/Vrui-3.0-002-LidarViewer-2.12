97  /* Calculate the projection coefficients: */
98  double p=e*Math::sin(geodetic[1]);
99  double t=Math::tan(Math::Constants<double>::pi*0.25-geodetic[1]*0.5)/Math::pow((1.0-p)/(1.0+p),0.5*e);

101  double theta=n*(geodetic[0]-lng0);
102 
103  /* Calculate the Lambert conformal map coordinates: */

115  /* Calculate the unprojection coefficients: */

117  double rho0y=rho0-(map[1]-offset[1])*unitFactor;
118  double rho=Math::copysign(Math::sqrt(Math::sqr(x)+Math::sqr(rho0y)),n);
