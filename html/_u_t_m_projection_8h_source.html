105  double nf=2.0*Math::sqrt(n)/(1.0+n);
106  double slat=Math::sin(geodetic[1]);
107  double t=Math::sinh(Math::atanh(slat)-nf*Math::atanh(nf*slat));
108  double etap=Math::atanh(Math::sin(geodetic[0]-lng0)/Math::sqrt(1.0+t*t));
