264 template <>
265 class Marshaller<std::string>
266  {
267  /* Methods: */
268  public:
269  static size_t getSize(const std::string& value)
270  {
271  return sizeof(UInt32)+value.length()*sizeof(std::string::value_type);
272  }
273  template <class DataSinkParam>
274  static void write(const std::string& value,DataSinkParam& sink)
275  {
276  sink.template write<UInt32>(UInt32(value.length()));
277  sink.template write<std::string::value_type>(value.data(),value.length());
278  }
279  template <class DataSourceParam>
280  static std::string read(DataSourceParam& source)
281  {
282  std::string result;
283  size_t length=source.template read<UInt32>();
284  result.reserve(length);
285  while(length>0)
286  {
287  const size_t bufferSize=256;
288  std::string::value_type buffer[bufferSize];
289  size_t readSize=length;
290  if(readSize>bufferSize)
291  readSize=bufferSize;
292  source.template read<std::string::value_type>(buffer,readSize);
293  result.append(buffer,buffer+readSize);
294  length-=readSize;
295  }
296 
297  return result;
298  }
299  };
