96  /* Calculate the projection coefficients: */

98  double q=(1-e2)*(p/(1.0-p*p)-0.5*Math::log((1.0-p)/(1.0+p)))/e;
99  double rho=radius*Math::sqrt(c-n*q)/n;

101 
102  /* Calculate the Albers equal area map coordinates: */

104  Scalar((rho0-rho*Math::cos(theta))/unitFactor+offset[1]));
