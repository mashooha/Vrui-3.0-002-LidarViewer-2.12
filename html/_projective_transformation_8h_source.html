130  {
131  double sub[3][2];
132  sub[0][0]=double(m(1,1))*double(m(2,2))-double(m(2,1))*double(m(1,2));
133  sub[0][1]=double(m(1,2))*double(m(2,0))-double(m(2,2))*double(m(1,0));
134  sub[1][0]=double(m(2,1))*double(m(0,2))-double(m(0,1))*double(m(2,2));
135  sub[1][1]=double(m(2,2))*double(m(0,0))-double(m(0,2))*double(m(2,0));
136  sub[2][0]=double(m(0,1))*double(m(1,2))-double(m(1,1))*double(m(0,2));
137  double det=double(m(0,0))*sub[0][0]+double(m(1,0))*sub[1][0]+double(m(2,0))*sub[2][0];
138 
139  return Vector<ScalarParam,2>(ScalarParam((sub[0][0]*double(v[0])+sub[1][0]*double(v[1]))/det),
140  ScalarParam((sub[0][1]*double(v[0])+sub[1][1]*double(v[1]))/det));
141  }
