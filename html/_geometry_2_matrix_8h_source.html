115  }
116  inline static ComponentArray<ScalarParam,2> divide(const ComponentArray<ScalarParam,2>& ca,const ScalarParam m[2][2])
117  {
118  double det=double(m[0][0])*double(m[1][1])-double(m[1][0])*double(m[0][1]);
119  return ComponentArray<ScalarParam,2>(ScalarParam((double(m[1][1])*double(ca[0])-double(m[0][1])*double(ca[1]))/det),
120  ScalarParam((double(m[0][0])*double(ca[1])-double(m[1][0])*double(ca[0]))/det));
121  }
122  inline static double determinant(const ScalarParam m[2][2])
123  {
124  return double(m[0][0])*double(m[1][1])-double(m[1][0])*double(m[0][1]);
125  }
126  inline static void invert(const ScalarParam m[2][2],ScalarParam result[2][2])
127  {
128  double det=double(m[0][0])*double(m[1][1])-double(m[1][0])*double(m[0][1]);
129  ScalarParam r11=ScalarParam(m[0][0]/det);
130  result[0][0]=ScalarParam(m[1][1]/det);
131  result[0][1]=ScalarParam(-m[0][1]/det);
132  result[1][0]=ScalarParam(-m[1][0]/det);
133  result[1][1]=r11;
134  }
