176  {
177  Scalar resultLambda=Math::Constants<Scalar>::max;
178  typename HitResult::Direction resultDirection=HitResult::INVALID_DIRECTION;
179  typename HitResult::Part resultPart=HitResult::INVALID_PART;
180  bool checkCaps=true;
181  Scalar lambda;
182 
183  Vector op1=ray.getOrigin()-p1;
184  Scalar dira=ray.getDirection()*axis;
185  Scalar op1a=op1*axis;
186 
187  /* Intersect ray with cylinder mantel: */
188  Scalar a=sqr(ray.getDirection())-Math::sqr(dira);
189  if(a!=Scalar(0))
190  {
191  Scalar b=op1*ray.getDirection()-op1a*dira;
192  Scalar c=sqr(op1)-Math::sqr(op1a)-sqrRadius;
193  Scalar det=Math::sqr(b)-c*a;
194  if(det>=Scalar(0))
195  {
196  det=Math::sqrt(det);
197  if((lambda=(-b-det)/a)>=Scalar(0))
198  {
199  /* Check if the intersection is valid: */
200  Scalar beta=op1a+dira*lambda;
201  if(beta>=Scalar(0)&&beta<=height)
202  {
203  resultLambda=lambda;
204  resultDirection=HitResult::ENTRY;
205  resultPart=HitResult::MANTEL;
206  }
207  }
208  else if((lambda=(-b+det)/a)>=Scalar(0))
209  {
210  /* Check if the intersection is valid: */
211  Scalar beta=op1a+dira*lambda;
212  if(beta>=Scalar(0)&&beta<=height)
213  {
214  resultLambda=lambda;
215  resultDirection=HitResult::EXIT;
216  resultPart=HitResult::MANTEL;
217  }
218  }
219  }
220  else

222  }
223

227  lambda=-op1a/dira;

241  }
