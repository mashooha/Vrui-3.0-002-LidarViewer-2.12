91  PPoint geodeticToMap(const PPoint& geodetic) const // Converts a 2D point in geodetic (longitude, latitude) coordinates to map coordinates
92  {
93  /*******************************************************************
94  These formulae are from the literature. Don't ask me to explain
95  them.
96  *******************************************************************/
97 
98  /* Calculate the projection coefficients: */
99  double sphi=Math::sin(double(geodetic[1]));
100  double sphi2=Math::sqr(sphi);
101  double cphi=Math::cos(double(geodetic[1]));
102  double cphi2=Math::sqr(cphi);
103  double N=radius/Math::sqrt((1.0-e2*sphi2));
104  double T=sphi2/cphi2;
105  double C=ep2*cphi2;
106  double A=(double(geodetic[0])-lng0)*cphi;
107  double M=(Mc1*double(geodetic[1])-Mc2*Math::sin(2.0*double(geodetic[1]))+Mc3*Math::sin(4.0*double(geodetic[1]))-Mc4*Math::sin(6.0*double(geodetic[1])))*radius;
108 
109  /* Calculate the transverse Mercator coordinates: */
110  double A2=Math::sqr(A);
111  return PPoint(Scalar(((1.0+((1.0-T+C)+(5.0-18.0*T+T*T+72.0*C-58.0*ep2)*A2/20.0)*A2/6.0)*A)*k0*N+offset[0]),
112  Scalar((M-M0+((1.0+((5.0-T+9.0*C+4.0*C*C)+(61.0-58.0*T+T*T+600.0*C-330.0*ep2)*A2/30.0)*A2/12.0)*A2/2.0)*N*sphi/cphi)*k0+offset[1]));
113  }

116  {

120  *******************************************************************/
121 
122  /* Calculate the reverse projection coefficients: */
123  double M=M0+(double(map[1])-offset[1])/k0;
124  double mu=M/IMc0;
125  double phi=mu+IMc1*Math::sin(2.0*mu)+IMc2*Math::sin(4.0*mu)+IMc3*Math::sin(6.0*mu)+IMc4*Math::sin(8.0*mu);
126  double sphi=Math::sin(phi);
127  double sphi2=Math::sqr(sphi);
128  double cphi=Math::cos(phi);
129  double cphi2=Math::sqr(cphi);
130  double kappa=1.0-e2*sphi2;
131  double N=radius/Math::sqrt(kappa);
132  double NbyR=kappa/(1.0-e2);
133  double T=sphi2/cphi2;
134  double C=ep2*cphi2;
135  double D=(double(map[0])-offset[0])/(N*k0);
136 
137  /* Calculate the geodetic coordinates: */

139  return PPoint(Scalar(lng0+((((5.0+(-3.0*C-2.0)*C+(24.0*T+28.0)*T+8.0*ep2)/120.0*D2-(1.0+C+2.0*T)/6.0)*D2+1.0)*D)/cphi),
