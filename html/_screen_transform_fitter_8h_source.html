112  {

115  /*******************************************************************
116  Calculate the distance vector between the transformed source point
117  and the target point. The transformation is spelled out in order to
118  reuse the intermediate results for the derivative calculation.
119  *******************************************************************/
120 
121  /* Calculate the non-uniform scaling: */
122  s[0]*=size[0];
123  s[1]*=size[1];
124 
125  /* Calculate the first rotation part: */
126  Scalar rX=q[1]*s[2]-q[2]*s[1]+q[3]*s[0];
127  Scalar rY=q[2]*s[0]-q[0]*s[2]+q[3]*s[1];
128  Scalar rZ=q[0]*s[1]-q[1]*s[0]+q[3]*s[2];
129  Scalar rW=q[0]*s[0]+q[1]*s[1]+q[2]*s[2];
130 
131  /* Calculate the second rotation part and the translation and difference: */
132  Vector d;
133  d[0]=rZ*q[1]-rY*q[2]+rW*q[0]+rX*q[3]+t[0]-tp[index][0];
134  d[1]=rX*q[2]-rZ*q[0]+rW*q[1]+rY*q[3]+t[1]-tp[index][1];
135  d[2]=rY*q[0]-rX*q[1]+rW*q[2]+rZ*q[3]+t[2]-tp[index][2];
136 
137  /* Calculate the difference magnitude: */
138  Scalar dist=Geometry::mag(d);
139 
140  Derivative result;
141 
142  /* Calculate the translational partial derivatives: */
143  result[0]=d[0]/dist;
144  result[1]=d[1]/dist;
145  result[2]=d[2]/dist;
146 
147  /* Calculate the rotational partial derivatives: */
148  result[3]=Scalar(2)*(+d[0]*rW-d[1]*rZ+d[2]*rY)/dist;
149  result[4]=Scalar(2)*(+d[0]*rZ+d[1]*rW-d[2]*rX)/dist;
150  result[5]=Scalar(2)*(-d[0]*rY+d[1]*rX+d[2]*rW)/dist;
151  result[6]=Scalar(2)*(+d[0]*rX+d[1]*rY+d[2]*rZ)/dist;
152 
153  /* Calculate the scaling partial derivatives: */
154  result[7]=((+q[0]*q[0]-q[1]*q[1]-q[2]*q[2]+q[3]*q[3])*d[0]
155  +(+q[0]*q[1]+q[1]*q[0]+q[2]*q[3]+q[3]*q[2])*d[1]
156  +(+q[0]*q[2]-q[1]*q[3]+q[2]*q[0]-q[3]*q[1])*d[2])*s[0]/dist;
157  result[8]=((+q[0]*q[1]+q[1]*q[0]-q[2]*q[3]-q[3]*q[2])*d[0]
158  +(-q[0]*q[0]+q[1]*q[1]-q[2]*q[2]+q[3]*q[3])*d[1]
159  +(+q[0]*q[3]+q[1]*q[2]+q[2]*q[1]+q[3]*q[0])*d[2])*s[1]/dist;
