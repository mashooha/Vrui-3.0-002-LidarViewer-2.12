94  bool isRidge(void) // Returns true if the query point is a ridge point
95  {
96  if(pcas[0].getNumPoints()>=2&&pcas[1].getNumPoints()>=2)
97  {
98  double radius=Math::sqrt(radius2);
99 
100  /* Calculate eigenvalues of both PCAs: */
101  double evs[2][2];
102  unsigned int numEvs[2];
103  bool isRidges[2];
104  for(int i=0;i<2;++i)
105  {
106  pcas[i].calcCovariance();
107  numEvs[i]=pcas[i].calcEigenvalues(evs[i]);
108  isRidges[i]=numEvs[i]==2&&Math::abs(evs[i][0])>=radius*0.75&&Math::abs(evs[i][1])<=radius*0.333;
109  }
110 
111  if(isRidges[0]&&!isRidges[1])
112  {
113  ATransform::Point qp=toPlane.transform(queryPoint);
114  return pcas[0].calcEigenvector(evs[0][1])*(Geometry::PCACalculator<2>::Point(qp[0],qp[1])-pcas[0].calcCentroid())<=1.0;
115  }
116  else if(!isRidges[0]&&isRidges[1])
117  {
118  ATransform::Point qp=toPlane.transform(queryPoint);
119  return pcas[1].calcEigenvector(evs[1][1])*(Geometry::PCACalculator<2>::Point(qp[0],qp[1])-pcas[1].calcCentroid())<=1.0;
120  }
121  else

123  }
