107  template <class TraversalFunctionParam>
108  void traverseTree(const Point& nodeCenter,Scalar nodeSize,int nodeDepth,TraversalFunctionParam& traversalFunction) const // Traverses tree in prefix order and calls traversal function for each node
109  {
110  /* Call traversal function: */
111  traversalFunction(nodeCenter,nodeSize,nodeDepth);
112 
113  if(!leaf)
114  {
115  /* Traverse all children: */
116  Scalar childNodeSize=Math::div2(nodeSize);
117  for(int i=0;i<numChildren;++i)
118  {
119  /* Calculate the child node's center: */
120  Point childNodeCenter=nodeCenter;
121  for(int j=0;j<dimension;++j)
122  if(i&(1<<j))
123  childNodeCenter[j]+=childNodeSize;
124  else
125  childNodeCenter[j]-=childNodeSize;
126 
127  /* Traverse the child node's subtree: */
128  children->children[i].traverseTree(childNodeCenter,childNodeSize,nodeDepth+1,traversalFunction);
129  }
130  }
