57 class RadiusNormalCalculator:public NormalCalculator // Class to calculate normal vectors by accumulating points from a neighborhood of fixed size
58  {
59  /* Elements: */
60  private:
61  Scalar radius2; // Squared search radius around query point
62  Point queryPoint; // The query point for which to calculate a plane equation
63  double pxpxs,pxpys,pxpzs,pypys,pypzs,pzpzs,pxs,pys,pzs; // Accumulated components of covariance matrix
64  size_t numPoints; // Number of accumulated points
65 
66  /* Constructors and destructors: */
67  public:
68  RadiusNormalCalculator(Scalar sRadius); // Creates normal calculator for the given search radius
69 
70  /* Methods: */
71  void operator()(const LidarPoint& lp) // Process the given LiDAR point
72  {
73  /* Check if the point is inside the search radius: */
74  if(Geometry::sqrDist(lp,queryPoint)<=radius2)
75  {
76  /* Accumulate the node point: */
77  pxpxs+=double(lp[0])*double(lp[0]);
78  pxpys+=double(lp[0])*double(lp[1]);
79  pxpzs+=double(lp[0])*double(lp[2]);
80  pypys+=double(lp[1])*double(lp[1]);
81  pypzs+=double(lp[1])*double(lp[2]);
82  pzpzs+=double(lp[2])*double(lp[2]);
83  pxs+=double(lp[0]);
84  pys+=double(lp[1]);
85  pzs+=double(lp[2]);
86 
87  ++numPoints;
88  }
89  }
90  const Point& getQueryPoint(void) const
91  {
92  return queryPoint;
93  }

96  return radius2;
