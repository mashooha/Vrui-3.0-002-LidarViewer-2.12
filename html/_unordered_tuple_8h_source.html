32 template <class ElementParam,int dimensionParam>
33 class UnorderedTuple
34  {
35  /* Embedded classes: */
36  public:
37  typedef ElementParam Element; // Type of tuple elements
38  static const int dimension=dimensionParam; // Dimension of the tuple
39 
40  /* Elements: */
41  private:
42  Element elements[dimension]; // The tuple's elements, sorted in ascending order
43 
44  /* Private methods: */
45  void sortTuple(void) // Sorts the tuple after a change
46  {
47  /* Simple insertion sort; dimensions are expected to be small: */
48  for(int test=1;test<dimension;++test)
49  {
50  Element testElement=elements[test];
51  int insertPos;
52  for(insertPos=test;insertPos>0&&testElement<elements[insertPos-1];--insertPos)
53  elements[insertPos]=elements[insertPos-1];
54  elements[insertPos]=testElement;
55  }
56  }
57 
58  /* Constructors and destructors: */
59  public:
60  UnorderedTuple(void) // Dummy constructor
61  {
62  }
63  UnorderedTuple(const Element sElements[dimensionParam]) // Construction from C-style array
64  {
65  /* Copy the source array: */
66  for(int i=0;i<dimension;++i)
67  elements[i]=sElements[i];
68 
69  /* Sort the elements: */
70  sortTuple();
71  }
72 
73  /* Methods: */
74  const Element* getElements(void) const // Returns element array
75  {
76  return elements;
77  }
78  Element operator[](int index) const // Returns element as rvalue
79  {
80  return elements[index];
81  }
82  static size_t hash(const UnorderedTuple& source,size_t tableSize) // Calculates a hash function for the given tuple and table size
83  {
84  size_t result=size_t(source.elements[0]);
85  for(int i=1;i<dimension;++i)
86  result=result*10000003+size_t(source.elements[i]);
87  return result%tableSize;
88  }
