316  void growTable(size_t newTableSize) // Grows the table without deleting current entries
317  {
318  /* Allocate new hash buckets: */
319  HashBucket* newHashBuckets=new HashBucket[newTableSize];
320 
321  /* Move all entries to the new table: */
322  for(size_t i=0;i<tableSize;++i)
323  {
324  /* Move all items in this hash bucket to the new table: */
325  while(hashBuckets[i].firstItem!=0)
326  {
327  /* Remove item from old hash bucket: */
328  HashBucketItem* item=hashBuckets[i].firstItem;
329  hashBuckets[i].firstItem=item->succ;
330 
331  /* Calculate item's new hash bucket index: */
332  size_t newBucketIndex=HashFunction::hash(item->getSource(),newTableSize);
333 
334  /* Add item to new hash bucket: */
335  item->succ=newHashBuckets[newBucketIndex].firstItem;
336  newHashBuckets[newBucketIndex].firstItem=item;
337  }
338  }
339 
340  /* Install the new hash table: */
341  tableSize=newTableSize;
342  delete[] hashBuckets;
343  hashBuckets=newHashBuckets;
344  maxEntries=(size_t)(tableSize*waterMark);
345  }
